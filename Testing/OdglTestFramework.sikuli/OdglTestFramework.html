from sikuli import *

import os
import signal
import subprocess
import platform
import threading

def StartTest(test, working_dir):
    dname = os.path.abspath(getBundlePath() + "/../../build/bin/")

    command = []
    if(os.path.isfile("/opt/VirtualGL/bin/vglrun")):
        command.append("vglrun")
    command.extend(["./run_test.sh", test])

    os.chdir(working_dir)
    return subprocess.Popen(command, stderr=subprocess.STDOUT, stdout=subprocess.PIPE)

def EndTest(process):
    out, err = process.communicate()
    if(float(out) < 0.016):
        print("Failure")
        exit(3)
    else:
        print("Success")
        exit(0)
